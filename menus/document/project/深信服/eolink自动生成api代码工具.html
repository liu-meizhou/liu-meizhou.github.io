eolink代码自动生成工具 ​

整体就是调用eolink的open api获取接口数据信息，通过这些信息用nodejs写文件写出ts类型和接口

  1. 在eolink平台拿到api的token
  2. 调用获取接口列表的接口，得到如下包括接口url、接口的request和response等数据结构
  3. 使用nodejs的writeFile把接口信息写入文件写入文件
  4. 最终生成比如axios.getXXX或者postXXX这样的接口，然后接口的参数类型和返回类型都是自动生成的
ts
interface Data {
    title: string; // 字段名称
    type: String;  // 字段类型
    description: String; // 描述说明
    enum: String[]; // 字段类型枚举的值
    property?: Data[]; // 如果字段类型是object
}
interface ListItem {
    url: string;
    method: string;
    description: string;
    req: Data[];
    res: Data[];
}
interface Res {
    list: Array<ListItem>
}
